291  return nullptr;
292  }
293  SDL_SetRenderTarget(sdl_ren, new_texture);
294  //przezroczysta tekstura
295  SDL_SetTextureBlendMode(new_texture, SDL_BLENDMODE_BLEND);
296  SDL_SetRenderDrawColor(sdl_ren, 0, 0, 0, 0);

305  //obrazek pierwotny

307  sdl_error("blend_texture: SDL_RenderCopy");
308  return nullptr;
309  }
310  //nałożenie półprzezroczystej maski
311  SDL_Rect dest;
312  dest.x = 0;
313  dest.y = 0;
314  dest.w = w;
315  dest.h = h;
316  SDL_SetRenderDrawColor(sdl_ren, color_blend.r, color_blend.g, color_blend.b, 255);
317  SDL_RenderFillRect(sdl_ren, &dest);
318  //przywrócenie rysowania na ekranie i trybu przenikania
319  SDL_SetRenderDrawBlendMode(sdl_ren, blend_mode);
