308 
309  int max_send_size = Config::buffer_size;

312  bool wynik1 = sendData(connections[sindex]->socket, msg2, max_send_size);
313  bool wynik2 = sendData(connections[sindex]->socket, msg2 + max_send_size,

317  }
318  }

329  retval = send(connections[sindex]->socket, msg2, (size_t) len, 0);
330  } else {

333  return false;
334  }
335  }
