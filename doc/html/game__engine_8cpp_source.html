76 void GameEngine::logic(volatile int& logic_cycles) {
77  if (logic_cycles > Config::logic_cycles_critical) {
78  Log::criticalError(
79  "Przekroczono liczbę niewykonanych cykli logicznych. Komputer nie nadąża z obliczeniami.");
80  app->exit();
81  logic_cycles = -1;
82  return;
83  }
84  cycles++;
85  while (SDL_PollEvent(&event) && !app->exiting()) { //pętla komunikatów SDL
86  if (event.type == SDL_QUIT) {
87  app->exit();
88  }
89  if (event.type == SDL_KEYDOWN) {
90  keyboard_event(event.key.keysym);
91  }
92  if (event.type == SDL_WINDOWEVENT) {
93  if (event.window.event == SDL_WINDOWEVENT_MOVED) {
94  Log::debug(
95  "Zmiana położenia okna (zatrzymana aplikacja) - reset licznika logicznego");
96  logic_cycles = 1;
97  }
98  if (event.window.event == SDL_WINDOWEVENT_RESTORED) {
99  Log::debug("Okno przywrócone.");
100  graphics->reload_textures();
101  }
102  }
103 
104  }
105  if (menu > 0) {
106  logic_menu();
107  } else {
108  logic_game();
109  }
110  logic_cycles--;
111 }
